165 template<class T> GOAT::maths::Vector<std::complex<double> > Trafo<T>::integrate(double t, std::vector<gridEntry> vge, double omegastart, double omegastop, int nsteps)
166 {
167 double k0;
168 GOAT:maths::Vector<std::complex<double> > E;
169 std::complex<double> phase;
170 double omega0 = 2.0 * M_PI * C_LIGHT_MU / tp.wvl;
171 double omega;
172 double domega = (omegastop - omegastart) / ((double)nsteps - 1.0);
173 double weight;
174 for (auto ge : vge) // Loop over all rays which hit the cell
175 {
176 for (int iomega = 0; iomega < nsteps; iomega++)
177 {
178 omega = omegastart + iomega * domega;
179 k0 = omega / C_LIGHT_MU;
180 phase = exp(I * calcPhase(ge.step, k0));
181 phase *= exp(-I * omega * t);
182 weight = exp(-twoSigma2 * (omega - omega0) * (omega - omega0));
183 E += weight * ge.E * phase;
184 }
185 }
186 return E;
