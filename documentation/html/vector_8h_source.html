359 {
360  Vector<T> y,a;
361 
362  a[0] = x[0];
363  a[1] = x[1];
364  a[2] = x[2];
365  y[1] = sqrt(a[0] * a[0] + a[1] * a[1] + a[2] * a[2]);
366  if (a[2] == 0.0)
367  if (y[0] == 0.0)
368  y[1] = 0;
369  else
370  y[1] = M_PI / 2.0;
371  else
372  {
373  y[1] = atan(sqrt(a[0] * a[0] + a[1] * a[1]) / a[2]);
374  if (a[2] < 0.0)
375  y[1] = M_PI + y[1];
376  }
377  if (a[0] == 0.0)
378  if (a[1] == 0.0)
379  y[2] = 0.0;
380  else
381  if (a[1] > 0.0)
382  y[2] = M_PI / 2.0;
383  else
384  y[2] = 3.0 * M_PI / 2.0;
385  else
386  {
387  y[2] = atan(a[1] / a[0]);
388  if (a[0] < 0.0)
389  y[2] = M_PI + y[2];
390  }
391  return y;
392 }
