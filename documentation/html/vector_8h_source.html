362 template <class T> Vector<T> cart2sphere(const Vector<T>& x)
363 {
364 Vector<T> y, a;
365
366 a[0] = x[0];
367 a[1] = x[1];
368 a[2] = x[2];
369 y[1] = std::sqrt(a[0] * a[0] + a[1] * a[1] + a[2] * a[2]);
370 if (a[2] == 0.0)
371 if (y[0] == 0.0)
372 y[1] = 0;
373 else
374 y[1] = M_PI / 2.0;
375 else
376 {
377 y[1] = atan(std::sqrt(a[0] * a[0] + a[1] * a[1]) / a[2]);
378 if (a[2] < 0.0)
379 y[1] = M_PI + y[1];
380 }
381 if (a[0] == 0.0)
382 if (a[1] == 0.0)
383 y[2] = 0.0;
384 else
385 if (a[1] > 0.0)
386 y[2] = M_PI / 2.0;
387 else
388 y[2] = 3.0 * M_PI / 2.0;
389 else
390 {
391 y[2] = atan(a[1] / a[0]);
392 if (a[0] < 0.0)
393 y[2] = M_PI + y[2];
394 }
395 return y;
396 }
